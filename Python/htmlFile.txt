	function toggleMobileMainMenu() {
		if ($('#mobileSearch').css('display') == 'none'||$('#mobileSearch').css('display')== null)
			toggleMobileMainNavigation();
		$('#mobileMainMenu').toggle();
		$('#mobileSearch').hide();

		if ($('#mobileMainMenu').css('display') == 'none'||$('#mobileMainMenu').css('display')== null)
			toggleMobileMainNavigation();
		$('#mobileSearch').toggle();
		$('#mobileMainMenu').hide();
	}
	
	
	function toggleMobileMainNavigation() {

			$("#hubLandingContainerDiv").toggle();
